prog := (stmt)* expr

stmt := "let" ident "=" expr ";"

expr := if
if := or | "if (" or "){" or "} else {" or "}" 
or := and ("||" and)*
and := equ ("&&" equ)*
equ := rel ("==" rel | "!=" rel)*
rel := add ("<" add | ">" add | ">=" add | "<=" add)*
add := mul ('+' mul | '-' mul)*
mul := unary ('*' unary | '/' unary)*
unary := ("-" | "!")? primary
primary := int | bool | ident | '(' expr ')'

bool := "true" | "false"
